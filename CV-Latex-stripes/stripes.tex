\newcommand{\portionheader}[1]{
  \begin{tikzpicture}[remember picture,overlay]
   \node [rectangle, fill=ggray, anchor=north, minimum width=\paperwidth, minimum height=5cm, text centered, align=left, inner ysep=.8cm] 
   (header) at (current page.north){};

	 \node[above=0.75cm of header.center] (name) {
		\LARGE{\textbf{#1}}
	};
	
	\node[left=5.75cm of header.center] (pic)  {
    		\includegraphics[draft]{picture}
    }; 	
     	
	\node[below left=0.25cm and -5.5cm of name] (info) {
		\small{\begin{tabular}{cl cl}
			 \faHome & I live here & \faMobilePhone & 0000000 \\
			 \faMailReply & \href{mailto:mail@mail}{\tt{mail@mail}} & \faPhone & 0000001 \\
			 \faMailReplyAll & \href{mailto:mail2@mail}{\tt{mail2@mail}} & \faSkype & skype \\
			 \faLinkedinSign & linkedin & & \\ % \fraMendeley & francescosensi
        \end{tabular}}
	};

	\node [below right=0.5cm and 7.5cm of header.center, label={[align=right]
	\footnotesize\textit{First}\\
	\footnotesize\textit{Second}\\
	\footnotesize{--anonymous}}
	] (phrase) {};


 \end{tikzpicture}
 \noindent
}
        
\newcommand{\portion}[4]{% color, sec. name, previous sec., text
\begin{tikzpicture}[remember picture,overlay]
\node [rectangle, fill=#1, anchor=north, minimum width=\paperwidth, minimum height=5cm, text centered, align=left, inner ysep=.8cm] (#2) at (#3.south){
	\begin{varwidth}{18cm} 
		\begin{minipage}[t!]{0.2\textwidth}
			\begin{tikzpicture}
				\foreach \x in {0} 
					  \draw [black, thick, fill=black] (\x,0) circle (1);
			    \foreach \x in {0} 
					  \draw [white, thick] (\x,0) circle (0.95);
			    \node[white,overlay] at (0,2.5) {\scriptsize{\textbf{#2}}};
			\end{tikzpicture}
		\end{minipage}
	\begin{minipage}[t!]{0.8\textwidth}
		#4
	\end{minipage}        
	\end{varwidth}
};
\end{tikzpicture}
\noindent
}

	\begin{varwidth}{18cm} 		#4	\end{varwidth} };
\end{tikzpicture}
\noindent
